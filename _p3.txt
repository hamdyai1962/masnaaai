    saveData: function() {
        var s = this;
        var data = JSON.stringify({
            agents: s.agents,
            swarms: s.swarms,
            memories: s.memories,
            kbDocs: s.kbDocs,
            testResults: s.testResults,
            evoGenerations: s.evoGenerations,
            analytics: s.analytics
        });
        if (s.puterOK) {
            puter.kv.set('hf_agents', data).catch(function(e) {
                LOGGER.error('SAVE', 'Puter save fail: ' + e.message);
                try { localStorage.setItem('hf_agents', data); } catch(x) {}
            });
        } else {
            try { localStorage.setItem('hf_agents', data); } catch(e) {
                LOGGER.error('SAVE', 'LocalStorage fail: ' + e.message);
            }
        }
        s.updateCounts();
        EVENTBUS.emit('data-saved', {count: s.agents.length});
    },

    loadData: function() {
        var s = this;
        return new Promise(function(resolve) {
            var parse = function(raw) {
                try {
                    if (!raw) return;
                    var d = JSON.parse(raw);
                    if (d.agents) s.agents = d.agents;
                    if (d.swarms) s.swarms = d.swarms;
                    if (d.memories) s.memories = d.memories;
                    if (d.kbDocs) s.kbDocs = d.kbDocs;
                    if (d.testResults) s.testResults = d.testResults;
                    if (d.evoGenerations) s.evoGenerations = d.evoGenerations;
                    if (d.analytics) s.analytics = Object.assign(s.analytics, d.analytics);
                } catch(e) { LOGGER.error('LOAD', 'Parse fail: ' + e.message); }
            };
            if (s.puterOK) {
                puter.kv.get('hf_agents').then(function(v) {
                    parse(v); s.updateCounts(); resolve();
                }).catch(function() {
                    parse(localStorage.getItem('hf_agents')); s.updateCounts(); resolve();
                });
            } else {
                parse(localStorage.getItem('hf_agents')); s.updateCounts(); resolve();
            }
        });
    },

    updateCounts: function() {
        var el = document.getElementById('cMy');
        if (el) el.textContent = this.agents.length;
        var sa = document.getElementById('stA');
        if (sa) sa.textContent = this.agents.length;
        this.analytics.totalAgents = this.agents.length;
    },

                s.toast('🚀 تم النشر: ' + (site.url || site.subdomain));
                s.modal('<h2>🚀 تم النشر بنجاح!</h2><p style="font-size:14px;margin:14px 0">رابط المصنع: <a href="' + (site.url || '#') + '" target="_blank" style="color:var(--green)">' + (site.url || site.subdomain) + '</a></p>');
            }).catch(function(e) {
                s.toast('⚠️ النشر: ' + e.message, 'wn');
            });
        } else {
            this.toast('📋 لا يمكن النشر في الوضع المحلي — استخدم Puter', 'wn');
        }
    }
