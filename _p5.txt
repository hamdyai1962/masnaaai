    };
    this.agents.push(agent);
    this.saveData();
    this.toast('🏗️ تم بناء "' + name + '"!', 'ok');
    this.go('my');

F.doImport = function() {
    var s = this;
    var file = document.getElementById('impFile');
    if (!file || !file.files[0]) { s.toast('اختر ملفاً', 'er'); return; }
    var reader = new FileReader();
    reader.onload = function(e) {
        try {
            var data = JSON.parse(e.target.result);
            var imported = 0;
            var list = data.agents || (Array.isArray(data) ? data : [data]);
            for (var i = 0; i < list.length; i++) {
                var valid = SEC.validateAgent(list[i]);
                if (valid) { s.agents.push(valid); imported++; }
            }
            s.saveData();
            s.closeModal();
            s.toast('📥 تم استيراد ' + imported + ' وكيل', 'ok');
            s.go('my');
        } catch(err) { s.toast('❌ ملف غير صالح: ' + err.message, 'er'); }
    };
    reader.readAsText(file.files[0]);
};

F.doGenImage = function() {
    var s = this;
    var prompt = (document.getElementById('imgPrompt')||{}).value;
    if (!prompt) { s.toast('اكتب وصفاً', 'er'); return; }
    if (!s.puterOK) { s.toast('يجب الاتصال بـ Puter', 'er'); return; }
    var btn = document.getElementById('imgBtn');
    if (btn) { btn.disabled = true; btn.innerHTML = '<span class="spin"></span> جاري التوليد...'; }
    var model = (document.getElementById('imgModel')||{}).value || 'dall-e-3';
    puter.ai.txt2img(prompt, {model:model}).then(function(img) {
        var res = document.getElementById('imgResult');
        if (res && img) {
            var url = (typeof img === 'string') ? img : (img.url || img.src || '');
            if (url) res.innerHTML = '<img src="' + url + '" style="max-width:100%;border-radius:12px;margin-top:8px">';
            else res.innerHTML = '<p style="color:var(--text2)">تم التوليد ولكن لا يوجد رابط</p>';
        }
        if (btn) { btn.disabled = false; btn.innerHTML = '🎨 توليد'; }
    }).catch(function(e) {
        if (btn) { btn.disabled = false; btn.innerHTML = '🎨 توليد'; }
        s.toast('❌ فشل التوليد: ' + e.message, 'er');
    });
};

    var text = (document.getElementById('ttsText')||{}).value;

    if (this.puterOK && puter.ai.txt2speech) {
        puter.ai.txt2speech(text).then(function(audio) {
            var res = document.getElementById('ttsResult');
            if (res) res.innerHTML = '<audio controls autoplay style="width:100%"><source src="' + (audio.url||audio) + '"></audio>';
        }).catch(function() {
            var u = new SpeechSynthesisUtterance(text);
            u.lang = 'ar'; speechSynthesis.speak(u);
        });
    } else {
        var u = new SpeechSynthesisUtterance(text);
        u.lang = 'ar'; speechSynthesis.speak(u);
    }
};

F.startSTT = function() {
    var s = this;
    if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
        s.toast('المتصفح لا يدعم التعرف على الصوت', 'er'); return;
    }
    var SR = window.SpeechRecognition || window.webkitSpeechRecognition;
    var rec = new SR();
    rec.lang = 'ar';
    rec.continuous = false;
    rec.onresult = function(e) {
        var txt = e.results[0][0].transcript;
        var res = document.getElementById('sttResult');
        if (res) res.innerHTML = '<div class="cd" style="margin-top:8px"><strong>النتيجة:</strong> ' + SEC.sanitize(txt) + '</div>';
    };
    rec.onerror = function(e) { s.toast('خطأ: ' + e.error, 'er'); };
    rec.start();
    s.toast('🎤 جاري الاستماع...', 'in');
};

F.render_mem = function() {
    var s = this;
    var el = document.getElementById('p-mem');
    if (!el) return;

    if (s.agents.length === 0) {

        rows += '<tr><td>' + SEC.sanitize(a.icon) + ' ' + SEC.sanitize(a.name) + '</td>' +

            '<td>' + (a.xp||0) + '</td>' +
            '<td><span class="bdg bdg-g">' + s.getStage(a.xp||0) + '</span></td></tr>';
    }

    el.innerHTML = '<div class="cd-h" style="font-size:18px;margin-bottom:14px">🧠 ذاكرة الوكلاء</div>' +
        '<div class="cd"><table style="width:100%;border-collapse:collapse">' +
        '<tr style="border-bottom:1px solid var(--border)"><th style="text-align:right;padding:8px;font-size:12px;color:var(--text2)">الوكيل</th>' +
        '<th style="text-align:right;padding:8px;font-size:12px;color:var(--text2)">محادثات</th>' +
        '<th style="text-align:right;padding:8px;font-size:12px;color:var(--text2)">XP</th>' +
        '<th style="text-align:right;padding:8px;font-size:12px;color:var(--text2)">المرحلة</th></tr>' +
        rows + '</table></div>';
};

F.getStage = function(xp) {
    if (xp >= 1000) return '🌟 أسطورة';
    if (xp >= 500) return '🏆 خبير';
    if (xp >= 200) return '⚡ محترف';
    if (xp >= 50) return '📖 متعلم';
    return '🌱 مبتدئ';
};

F.render_kb = function() {
    var s = this;
    var el = document.getElementById('p-kb');
    if (!el) return;

    var docs = '';
    if (s.kbDocs.length === 0) {
        docs = '<div class="emy"><div class="ei">📚</div><h3>لا توجد وثائق</h3><p>أضف وثائق معرفة لتحسين إجابات وكلائك</p></div>';
    } else {
        for (var i = 0; i < s.kbDocs.length; i++) {
            var d = s.kbDocs[i];
            docs += '<div class="cd" style="margin-bottom:8px"><div style="display:flex;justify-content:space-between;align-items:center">' +
                '<strong>' + SEC.sanitize(d.title) + '</strong>' +
                '<button class="btn bd bxs" onclick="F.deleteKB(' + i + ')">🗑️</button></div>' +
                '<p style="font-size:11px;color:var(--text2);margin-top:4px">' + SEC.sanitize((d.content||'').substring(0,150)) + '...</p>' +
                '<div style="margin-top:4px"><span class="bdg bdg-b">' + SEC.sanitize(d.source||'غير محدد') + '</span></div></div>';
        }
    }

    el.innerHTML = '<div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:14px;flex-wrap:wrap;gap:8px">' +
        '<div class="cd-h" style="font-size:18px;margin:0;border:none;padding:0">📚 قاعدة المعرفة (' + s.kbDocs.length + ')</div>' +
        '<button class="btn bg bsm" onclick="F.addKB()">➕ إضافة وثيقة</button></div>' + docs;
};

F.addKB = function() {
    var s = this;
    s.modal(
        '<h2>📚 إضافة وثيقة معرفة</h2>' +
        '<div class="fg"><label class="lbl">العنوان *</label><input class="inp" id="kbTitle" placeholder="عنوان الوثيقة"></div>' +
        '<div class="fg"><label class="lbl">المحتوى *</label><textarea class="inp" id="kbContent" style="min-height:120px" placeholder="محتوى الوثيقة..."></textarea></div>' +
        '<div class="fg"><label class="lbl">المصدر</label><input class="inp" id="kbSource" placeholder="مصدر المعلومة"></div>' +
        '<div class="fg"><label class="lbl">كلمات مفتاحية</label><input class="inp" id="kbKeys" placeholder="مفصولة بفواصل"></div>' +
        '<button class="btn bg" onclick="F.saveKB()">💾 حفظ</button>'
    );
};

F.saveKB = function() {
    var title = (document.getElementById('kbTitle')||{}).value;
    var content = (document.getElementById('kbContent')||{}).value;
    if (!title || !content) { this.toast('أدخل العنوان والمحتوى', 'er'); return; }
    this.kbDocs.push({
        id: this.uid(),
        title: title,
        content: content,
        source: (document.getElementById('kbSource')||{}).value || '',
        keywords: ((document.getElementById('kbKeys')||{}).value || '').split(',').map(function(k){return k.trim()}).filter(function(k){return k}),
        createdAt: new Date().toISOString()
    });
    this.saveData();
    this.closeModal();
    this.toast('📚 تم إضافة الوثيقة', 'ok');
    this.go('kb');
};

F.deleteKB = function(idx) {
    this.kbDocs.splice(idx, 1);
    this.saveData();
    this.toast('تم حذف الوثيقة', 'in');
    this.go('kb');
};
